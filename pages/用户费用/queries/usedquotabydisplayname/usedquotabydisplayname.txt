SELECT
  l.*,u.display_name AS display_name
FROM
  public.logs l
JOIN
 users u
ON
 u.id=l.user_id
WHERE
  "display_name" ilike '%{{Table1.selectedRow.display_name}}%'
ORDER BY
  "{{data_table.sortOrder.column || 'id'}}" {{data_table.sortOrder.order !== "desc" ? "" : "DESC"}}
LIMIT
  {{data_table.pageSize}}
OFFSET
  {{data_table.pageOffset}}
